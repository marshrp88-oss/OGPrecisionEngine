Tier breakpoints are functions of the targets. Do not hardcode the breakpoints — compute them as shown.
typescriptfunction mrrPayoutGross(mrr: number, mrrTarget: number): number {
  if (mrr <= 0) return 0;
  const tier1Cap = 349.93;
  const tier2Cap = 489.93;
  const tier3Cap = mrrTarget - 0.07;  // = 699.93 when target = 700
  
  const tier1 = Math.max(0, Math.min(mrr, tier1Cap) - 0)          * 0.3705;
  const tier2 = Math.max(0, Math.min(mrr, tier2Cap) - tier1Cap)   * 0.9634;
  const tier3 = Math.max(0, Math.min(mrr, tier3Cap) - tier2Cap)   * 5.5212;
  const tier4 = Math.max(0, mrr - tier3Cap)                       * 0.65;
  
  return tier1 + tier2 + tier3 + tier4;
}

function nrrPayoutGross(nrr: number, nrrTarget: number): number {
  if (nrr <= 0) return 0;
  const tier1Cap = 2999.40;
  const tier2Cap = 4199.40;
  const tier3Cap = nrrTarget - 0.6;  // = 5999.40 when target = 6000
  
  const tier1 = Math.max(0, Math.min(nrr, tier1Cap) - 0)          * 0.0204;
  const tier2 = Math.max(0, Math.min(nrr, tier2Cap) - tier1Cap)   * 0.0388;
  const tier3 = Math.max(0, Math.min(nrr, tier3Cap) - tier2Cap)   * 0.2801;
  const tier4 = Math.max(0, nrr - tier3Cap)                       * 0.042;
  
  return tier1 + tier2 + tier3 + tier4;
}

function commissionTakeHome(mrr: number, nrr: number, mrrTarget: number, nrrTarget: number, taxRate: number): number {
  const gross = mrrPayoutGross(mrr, mrrTarget) + nrrPayoutGross(nrr, nrrTarget);
  return gross * (1 - taxRate);
}

function commissionPayoutDate(salesMonth: Date): Date {
  // Payout is the 22nd of the month FOLLOWING the sales month
  const payoutYear = salesMonth.getMonth() === 11 ? salesMonth.getFullYear() + 1 : salesMonth.getFullYear();
  const payoutMonth = (salesMonth.getMonth() + 1) % 12;
  return new Date(payoutYear, payoutMonth, 22);
}

typescriptfunction safeToSpend(ctx: FinanceContext): number {
  const today = ctx.today;                      // e.g., 2026-04-21
  const nextPayday = effectivePayday(ctx.nextPaydayNominal);  // weekend-adjusted
  
  const billsInCycle = ctx.bills.filter(b =>
    b.include === true &&
    b.amount > 0 &&
    b.nextDueDate >= today &&
    b.nextDueDate < nextPayday    // STRICT less-than
  );
  const billsTotal = billsInCycle.reduce((s, b) => s + b.amount, 0);
  
  const oneTimeInCycle = ctx.oneTimeExpenses.filter(e =>
    e.paid === false &&
    e.dueDate !== null &&
    e.dueDate >= today &&
    e.dueDate <= nextPayday
  );
  const oneTimeTotal = oneTimeInCycle.reduce((s, e) => s + e.amount, 0);
  
  const requiredHold =
    billsTotal +
    ctx.pendingHolds +
    ctx.minimumCushion +
    ctx.checkingFloor +
    ctx.irregularBuffer +
    ctx.timingBuffer +

  return Math.max(0, ctx.checkingBalance - requiredHold);
}

typescriptfunction computeForwardReserve(ctx: FinanceContext): number {
  // Bills due 1st through 7th of NEXT month, Include=TRUE
  const nextMonth = addMonths(startOfMonth(ctx.today), 1);
  const nextMonthBillsFirst7 = ctx.bills
    .filter(b => b.include === true)
    .filter(b => b.dueDay >= 1 && b.dueDay <= 7)

  const sevenDaysVariable = 7 * dailyVariable;
  
  return nextMonthBillsFirst7 + sevenDaysVariable;

B6. Effective Payday (weekend adjustment) — already spec'd, verify it exists
typescriptfunction effectivePayday(nominal: Date): Date {
  const day = nominal.getDay();  // 0 = Sun, 6 = Sat
  if (day === 6) {
    // Saturday → Friday
    return addDays(nominal, -1);
  }
  if (day === 0) {
    // Sunday → Friday  
    return addDays(nominal, -2);
  }
  return nominal;
}
